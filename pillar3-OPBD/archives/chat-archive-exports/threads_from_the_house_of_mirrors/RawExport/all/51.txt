```ts
type Enforcement = {
  collab: boolean;
  halCount: number;
  errors: string[];
};

export function enforceAndTransform(src: string, renderHal: (i:number)=&gt;string) {
  const lines = src.split(/\r?\n/);
  const state: Enforcement = { collab: false, halCount: 0, errors: [] };

  const startIdx = lines.findIndex(l =&gt; l.trim() === "[We collab]");
  const endIdx   = lines.findIndex(l =&gt; l.trim() === "[End collab]");

  if (startIdx === -1) state.errors.push("Missing [We collab].");
  if (endIdx === -1)   state.errors.push("Missing [End collab].");
  if (startIdx !== -1 &amp;&amp; endIdx !== -1 &amp;&amp; endIdx &lt; startIdx)
    state.errors.push("[End collab] appears before [We collab].");

  state.collab = startIdx !== -1 &amp;&amp; endIdx !== -1 &amp;&amp; endIdx &gt; startIdx;

  if (!state.collab) return { ok:false, errors: state.errors, output: src };

  const body = lines.slice(startIdx+1, endIdx).join("\n");

  // enforce exact tag
  const halTag = /\[Hal talks here\]/g;
  const matches = body.match(halTag) || [];
  state.halCount = matches.length;

  if (state.halCount &gt; 8) state.errors.push("Too many HAL inserts (max 8).");

  // transform
  let i = 0;
  const transformed = body.replace(halTag, () =&gt; {
    i++;
    return renderHal(i); // you supply formatted block
  });

  const output =
    lines.slice(0, startIdx).join("\n") + "\n" +
    transformed + "\n" +
    lines.slice(endIdx+1).join("\n");

  return { ok: state.errors.length === 0, errors: state.errors, output };
}
```
